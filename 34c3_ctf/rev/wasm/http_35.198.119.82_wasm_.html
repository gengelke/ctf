  <script src="wasmblock.js"></script>
  <script>
    window.Module = {};
    fetchAndInstantiate("./flag.wasm", {})
    .then(mod => {
      Module.check = mod.exports.check;
      Module.alloc = mod.exports.alloc;
      Module.dealloc_str = mod.exports.dealloc_str;
      Module.memory = mod.exports.memory;
    })

    function check_flag() {
      let flag = document.getElementById("flag").value;
      let buf = newString(Module, flag);
      let result = document.getElementById("result");
      result.setAttribute("style", "");

      if (Module.check(buf)) {
        result.setAttribute("class", "alert alert-success");
        result.innerText = "Good job :-)";
      } else {
        result.setAttribute("class", "alert alert-danger");
        result.innerText = "Invalid flag :-(";
      }
    }
  </script>

  <div class="container" style="margin-top: 200px;">
    <form onsubmit="check_flag(); return false;">
    <div class="form-group">
      <input type="text" class="form-control" id="flag" placeholder="Enter flag">
    </div>
    <button type="submit" class="btn btn-primary">Submit</button>
    </form>
    <div id="result" class="alert" role="alert" style="display:none">
    </div>
